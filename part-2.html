Tryb konfiguracji

Tryb konfiguracji

                    
                        on evtStart do
                          TaskValueSet,3,1,1
                          if [Plugin#GPIO#Pinstate#14]=1
                            TimerSet,2,1
                          else
                            if [vcc#vcc]<3.20
                              DeepSleep,4294
                            endif
                            TaskValueSet,3,1,0
                          endif
                          TimerSet,3,30
                        endon
                    
                
                    
                        on System#Boot do AsyncEvent,evtStart endon
                        on System#Wake do AsyncEvent,evtStart endon
                        on Rules#Timer=2 do
                          TaskValueSet,3,1,[Plugin#GPIO#Pinstate#14]
                          TimerSet,4,600
                        endon
                        on Rules#Timer=3 do
                          AsyncEvent,evtNext
                        endon
                        on Rules#Timer=4 do
                          DeepSleep,900
                        endif
